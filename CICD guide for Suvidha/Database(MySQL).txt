create database suvidha;
use suvidha

CREATE TABLE `application` (
  `id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(255) DEFAULT NULL,
  `email` varchar(255) DEFAULT NULL,
  `age` int DEFAULT NULL,
  `address` varchar(255) DEFAULT NULL,
  `phone` bigint DEFAULT NULL,
  `application_date` date DEFAULT NULL,
  `nursing_type` varchar(255) DEFAULT NULL,
  `specialization` varchar(255) DEFAULT NULL,
  `joining_date` date DEFAULT NULL,
  `status` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `email` (`email`),
  KEY `specialization` (`specialization`),
  CONSTRAINT `application_ibfk_1` FOREIGN KEY (`specialization`) REFERENCES `departments` (`specialization`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `datewise_medicalhistory` (
  `id` int NOT NULL AUTO_INCREMENT,
  `medical_history_id` varchar(255) DEFAULT NULL,
  `history_keepup_date` date DEFAULT NULL,
  `medical_history_content` varchar(511) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `medical_history_id` (`medical_history_id`),
  CONSTRAINT `datewise_medicalhistory_ibfk_1` FOREIGN KEY (`medical_history_id`) REFERENCES `medical_history` (`medical_history_id`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `departments` (
  `id` int NOT NULL AUTO_INCREMENT,
  `specialization` varchar(255) DEFAULT NULL,
  `total_cases` bigint DEFAULT NULL,
  `case_live` bigint DEFAULT NULL,
  `availability` varchar(255) DEFAULT NULL,
  `case_solved` bigint DEFAULT NULL,
  `consultancy_fees` float DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `specialization` (`specialization`)
) ENGINE=InnoDB AUTO_INCREMENT=30 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `diseases` (
  `id` int NOT NULL AUTO_INCREMENT,
  `disease_name` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `disease_name` (`disease_name`),
  KEY `disease_name_2` (`disease_name`)
) ENGINE=InnoDB AUTO_INCREMENT=720 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `doctor` (
  `id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(255) DEFAULT NULL,
  `email` varchar(255) DEFAULT NULL,
  `age` int DEFAULT NULL,
  `address` varchar(255) DEFAULT NULL,
  `phone` bigint DEFAULT NULL,
  `specialization` varchar(255) DEFAULT NULL,
  `experience` float DEFAULT NULL,
  `position` varchar(255) DEFAULT NULL,
  `case_live` bigint DEFAULT NULL,
  `joining_date` date DEFAULT NULL,
  `salary` float DEFAULT NULL,
  `case_solved` bigint DEFAULT NULL,
  `solved` int DEFAULT NULL,
  `password` varchar(255) DEFAULT NULL,
  `type` varchar(255) DEFAULT NULL,
  `total_cases` bigint DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `email` (`email`),
  KEY `specialization` (`specialization`),
  CONSTRAINT `doctor_ibfk_1` FOREIGN KEY (`specialization`) REFERENCES `departments` (`specialization`)
) ENGINE=InnoDB AUTO_INCREMENT=15 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `doctor_position` (
  `id` int NOT NULL AUTO_INCREMENT,
  `position` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=11 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `feedback` (
  `id` int NOT NULL AUTO_INCREMENT,
  `feedback` varchar(1023) DEFAULT NULL,
  `name` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `medical_assignments` (
  `id` int NOT NULL AUTO_INCREMENT,
  `patient_email` varchar(255) DEFAULT NULL,
  `doctor_email` varchar(255) DEFAULT NULL,
  `medical_history_id` varchar(255) DEFAULT NULL,
  `nurses_emails` varchar(1023) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `medical_assignments_ibfk_1` (`medical_history_id`),
  CONSTRAINT `medical_assignments_ibfk_1` FOREIGN KEY (`medical_history_id`) REFERENCES `medical_history` (`medical_history_id`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `medical_history` (
  `id` int NOT NULL AUTO_INCREMENT,
  `medical_history_id` varchar(255) DEFAULT NULL,
  `medical_history_detail` varchar(1023) DEFAULT NULL,
  `medical_bill` float DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `unique_medical_history_id` (`medical_history_id`),
  KEY `medical_history_id` (`medical_history_id`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `messages` (
  `id` int NOT NULL AUTO_INCREMENT,
  `messages` varchar(255) DEFAULT NULL,
  `patient_email` varchar(255) DEFAULT NULL,
  `doctor_email` varchar(255) DEFAULT NULL,
  `time_send` datetime DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `nurse` (
  `id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(255) DEFAULT NULL,
  `email` varchar(255) DEFAULT NULL,
  `age` int DEFAULT NULL,
  `address` varchar(255) DEFAULT NULL,
  `phone` bigint DEFAULT NULL,
  `specialization` varchar(255) DEFAULT NULL,
  `experience` float DEFAULT NULL,
  `position` varchar(255) DEFAULT NULL,
  `cases_managed` bigint DEFAULT NULL,
  `joining_date` date DEFAULT NULL,
  `salary` float DEFAULT NULL,
  `type` varchar(255) DEFAULT NULL,
  `password` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `email` (`email`),
  KEY `specialization` (`specialization`),
  CONSTRAINT `nurse_ibfk_1` FOREIGN KEY (`specialization`) REFERENCES `departments` (`specialization`)
) ENGINE=InnoDB AUTO_INCREMENT=17 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `nursing_position` (
  `id` int NOT NULL AUTO_INCREMENT,
  `position` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=8 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `patients` (
  `id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(255) DEFAULT NULL,
  `email` varchar(255) DEFAULT NULL,
  `age` int DEFAULT NULL,
  `address` varchar(255) DEFAULT NULL,
  `phone` bigint DEFAULT NULL,
  `disease_name` varchar(255) DEFAULT NULL,
  `treatment_provided` varchar(511) DEFAULT NULL,
  `admission_date` date DEFAULT NULL,
  `discharge_date` date DEFAULT NULL,
  `medical_history_id` varchar(255) DEFAULT NULL,
  `symptoms` varchar(511) DEFAULT NULL,
  `appointment_day` date DEFAULT NULL,
  `password` varchar(255) DEFAULT NULL,
  `doctor_name` varchar(255) DEFAULT NULL,
  `type` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `medical_history_id` (`medical_history_id`),
  UNIQUE KEY `email` (`email`,`appointment_day`,`doctor_name`),
  KEY `disease_name` (`disease_name`),
  CONSTRAINT `patients_ibfk_1` FOREIGN KEY (`disease_name`) REFERENCES `diseases` (`disease_name`)
) ENGINE=InnoDB AUTO_INCREMENT=7 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `prescription_summary` (
  `id` int NOT NULL AUTO_INCREMENT,
  `pharmacist_email` varchar(255) DEFAULT NULL,
  `patient_email` varchar(255) DEFAULT NULL,
  `prescription_day` date DEFAULT NULL,
  `medical_history_id` varchar(255) DEFAULT NULL,
  `prescription_detail` varchar(255) DEFAULT NULL,
  `paid_status` varchar(255) DEFAULT NULL,
  `prescription_bill` float DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `prescription_day` (`prescription_day`,`medical_history_id`),
  KEY `medical_history_id` (`medical_history_id`),
  CONSTRAINT `prescription_summary_ibfk_1` FOREIGN KEY (`medical_history_id`) REFERENCES `medical_history` (`medical_history_id`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `shift` (
  `id` int NOT NULL AUTO_INCREMENT,
  `shifts` varchar(255) DEFAULT NULL,
  `shift_start` time DEFAULT NULL,
  `shift_end` time DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `staff` (
  `id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(255) DEFAULT NULL,
  `joining_date` date DEFAULT NULL,
  `salary` float DEFAULT NULL,
  `email` varchar(255) DEFAULT NULL,
  `age` int DEFAULT NULL,

  PRIMARY KEY (`id`),
  UNIQUE KEY `email` (`email`)
